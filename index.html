// Q1. Remove duplicates
const a = [2, 5, 7, 9, 0, 2, 4, 3, 7, 10, 1, 1, 2, 1, 12, 6]

let set = {}

let ans = []
for (let i = 0; i < a.length; i++) { // o(n)
    let key = a[i]
    if (!set[key]) {
        set = { ...set, [key]: i }
        ans.push(a[i])
    }
}
// console.log(ans)

let firstIdx = -1
let secondIdx = -1

// store the indexs
for (let i = 0; i < ans.length; i++) { // o(n)
    let key = ans[i]
    set = { ...set, [key]: i - 1 }
}

console.log(ans)

// Q2. find the indexs whose sum = 21
for (let i = 0; i < ans.length; i++) {
    if (set[21 - ans[i]]) {
        firstIdx = set[21 - ans[i]]
        secondIdx = i
        break
    }
}
